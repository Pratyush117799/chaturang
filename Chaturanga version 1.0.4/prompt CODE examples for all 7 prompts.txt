```typescript
// pages/index.tsx (Next.js)
import { useEffect, useRef, useState } from 'react';
import { gsap } from 'gsap';
import { Chessboard } from 'react-chessboard';
import * as THREE from 'three';
import { Canvas } from '@react-three/fiber';
import { OrbitControls } from '@react-three/drei';
import { createClient } from '@sanity/client';
import io from 'socket.io-client';
import Chart from 'chart.js/auto';
import { RemotionPlayer } from '@remotion/player'; // Assume installed

const sanityClient = createClient({ projectId: 'your-id', dataset: 'production', useCdn: true });

export default function Home() {
  const [content, setContent] = useState({ title: 'Chaturanga — the classic reborn', bullets: [] });
  const [activePlayers, setActivePlayers] = useState<number[]>([]);
  const heroRef = useRef<HTMLDivElement>(null);
  const chartRef = useRef<HTMLCanvasElement>(null);
  const threeRef = useRef<THREE.Scene>(null); // For 3D

  useEffect(() => {
    // Fetch dynamic content from Sanity
    sanityClient.fetch('*[_type == "homepage"]{title, bullets}').then(setContent).catch(console.error);

    // GSAP animations
    gsap.from(heroRef.current, { opacity: 0, y: 50, duration: 1, ease: 'power2.out' });

    // WebSockets for real-time active players
    const socket = io('your-backend-url');
    socket.on('activePlayers', (data: number[]) => setActivePlayers(data));

    // Chart.js dashboard
    if (chartRef.current) {
      new Chart(chartRef.current, {
        type: 'line',
        data: { labels: ['Now'], datasets: [{ label: 'Active Players', data: activePlayers }] },
        options: { responsive: true },
      });
    }

    // Cleanup
    return () => socket.disconnect();
  }, []);

  // 3D Board Preview Component
  const Board3D = () => (
    <Canvas>
      <mesh rotation={[-Math.PI / 2, 0, 0]}>
        <planeGeometry args={[8, 8]} />
        <meshStandardMaterial color="sandybrown" /> {/* Battlefield feel */}
      </mesh>
      <OrbitControls />
      <ambientLight />
    </Canvas>
  );

  return (
    <div className="grid grid-cols-2 bg-offwhite dark:bg-charcoal"> {/* Tailwind classes */}
      <div ref={heroRef}>
        <h1 className="text-4xl font-merriweather text-gold">{content.title}</h1>
        <ul>{content.bullets.map((b: string) => <li key={b}>{b}</li>)}</ul>
        <button className="bg-teal text-white">Play Now</button> {/* GSAP hover: onMouseEnter gsap.to(scale:1.05) */}
      </div>
      <div>
        {/* Remotion Video Loop */}
        <RemotionPlayer component={() => <div>Queen's Gambit Animation</div>} durationInFrames={300} compositionWidth={1920} compositionHeight={1080} />
        <Board3D /> {/* Immersive 3D preview */}
        <canvas ref={chartRef} /> {/* Real-time chart */}
      </div>
    </div>
  );
}
```

## Prompt 3 Expansion: Game UI and Play Functionality (React Example with chess.js)
**Detailed Explanation**: Expands with a GameBoard component, including drag animations (GSAP), Stockfish analysis (WebAssembly stub), real-time moves (Socket.IO), and 3D toggle. Operations: chess.js for validation/FEN; WebWorkers for WASM to avoid UI block. Nuances: Keyboard accessibility (arrows for moves); cultural particles (e.g., diya sparks on capture for Indian vibe). Implications: Smooth UX like Chess.com; handles disconnects (reconnect logic). Edge: Variant rules (e.g., Chaturanga no castling).

```typescript
// components/GameBoard.tsx
import { useState, useEffect } from 'react';
import Chess from 'chess.js';
import { Chessboard } from 'react-chessboard';
import { gsap } from 'gsap';
import io from 'socket.io-client';
// Assume Stockfish WASM loaded via worker
const worker = new Worker(new URL('./stockfish.worker.ts', import.meta.url));

const GameBoard = ({ gameId }: { gameId: string }) => {
  const [game, setGame] = useState(new Chess());
  const [position, setPosition] = useState(game.fen());
  const socket = io('backend-url');

  useEffect(() => {
    socket.emit('joinGame', gameId);
    socket.on('move', (move: string) => {
      game.move(move);
      setPosition(game.fen());
      // GSAP animation for move
      gsap.to('.piece-' + move, { x: 100, y: 100, duration: 0.2, ease: 'easeOut' }); // Simplified
    });

    worker.onmessage = (e) => console.log('Analysis:', e.data); // Best move from Stockfish

    return () => socket.disconnect();
  }, [gameId]);

  const onDrop = (source: string, target: string) => {
    const move = game.move({ from: source, to: target, promotion: 'q' });
    if (move) {
      socket.emit('move', move.san);
      setPosition(game.fen());
      worker.postMessage({ cmd: 'go', fen: position }); // Analyze
      return true;
    }
    return false;
  };

  return <Chessboard position={position} onPieceDrop={onDrop} />;
};

export default GameBoard;
```
